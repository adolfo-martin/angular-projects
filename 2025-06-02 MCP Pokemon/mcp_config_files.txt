# Instalación paso a paso:

# 1. Crear directorio para el proyecto
mkdir pokeapi-mcp
cd pokeapi-mcp

# 2. Crear entorno virtual (recomendado)
python -m venv venv

# En Windows:
venv\Scripts\activate
# En macOS/Linux:
source venv/bin/activate

# 3. Instalar dependencias
pip install aiohttp mcp

# 4. Guardar el código del servidor como pokeapi_mcp_server.py

# 5. Hacer el archivo ejecutable (Linux/macOS)
chmod +x pokeapi_mcp_server.py

# 6. Configurar Claude Desktop:
# - Localizar el archivo claude_desktop_config.json en tu sistema
# - Añadir la configuración del servidor MCP
# - Actualizar las rutas con las rutas reales de tu sistema

# 7. Reiniciar Claude Desktop

# 8. Probar la integración con comandos como:
# "Búscame información sobre Pikachu"
# "¿Cuáles son las estadísticas de Charizard?"
# "Explícame el tipo fuego"
